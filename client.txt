//задание: создать файл, который будет содержать фамилию студента и 4 оценки. 
//(например, 5 5 4 4) и сообщить, когда получает стипендию а когда нет.
//(обычная, повышенная стипендия) или сколько у студента задолженностей


//client
#include <iostream>
#include <fstream>
#include <string>
#include <sstream>
#include <windows.h>

using namespace std;

const string REQUEST_FILE = "D:\\REQUEST.bin";
const string ANSWER_FILE = "D:\\ANSWER.bin";
const int NAME_SIZE = 25;

struct Student {
    char lastName[NAME_SIZE];
    int grades[4];
};

struct ScholarshipInfo {
    int status;  // 0 - нет стипендии, 1 - обычная, 2 - повышенная
    int debtCount;
};


int main() {
    setlocale(LC_ALL, "ru");

    while (true) {
        Student student;

        cout << "Введите фамилию студента (до " << NAME_SIZE<< " символов): ";
        cin >> student.lastName;

        cout << "Введите 4 оценки студента (разделенные пробелами): ";
        for (int i = 0; i < 4; ++i) {
            cin >> student.grades[i];
        }

        ofstream requestFile(REQUEST_FILE, ios::binary | ios::app);
        requestFile.write((char*)&student, sizeof(Student));
        requestFile.close();

        cout << endl;
        cout << "Запрос отправлен. Ожидание ответа..." << endl;
        cout << endl;


        ifstream answerFile(ANSWER_FILE, ios::binary);
        ScholarshipInfo scholarshipInfo;

        answerFile.seekg(0, ios::end);

        long pred_size;

        pred_size = answerFile.tellg();

        while (pred_size >= answerFile.tellg()) {
            Sleep(100);
            answerFile.seekg(0, ios::end);
        }

        answerFile.seekg(pred_size, ios::beg);

        answerFile.read((char*)&scholarshipInfo, sizeof(ScholarshipInfo));
        answerFile.close();

        cout << "Ответ получен:" << endl;
        if (scholarshipInfo.status == 0) {
            cout << "Студент " << student.lastName << " не получает стипендию. Количество задолженностей: " << scholarshipInfo.debtCount << endl;
            cout << endl;
        }
        else if (scholarshipInfo.status == 1) {
            cout << "Студент " << student.lastName << " получает обычную стипендию." << endl;
            cout << endl;
        }
        else {
            cout << "Студент " << student.lastName << " получает повышенную стипендию!" << endl;
            cout<< endl;
        }

    }

    return 0;
}
